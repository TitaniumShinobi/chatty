1. Add a generic “X ###:” heading normalizer
In your text utils, extend the heading logic:

ts
// existing
const PSEUDO_HEADING_PATTERN =
  /^={0,3}\s*([A-Z][A-Z0-9 _#:-]+?)\s*#{0,3}\s*:?\s*$/;

// new, specifically for trailing ###:
const HASH_HEADING_PATTERN =
  /^([A-Z][A-Za-z0-9 _-]+)\s+#{3}\s*:?\s*(?:\((continued)\))?\s*$/;

function normalizeHashHeadings(line: string): string {
  const match = line.match(HASH_HEADING_PATTERN);
  if (!match) return line;

  const base = match[1].trim();
  const isContinued = !!match[2];

  const title =
    (base.charAt(0).toUpperCase() + base.slice(1).toLowerCase()) +
    (isContinued ? " (continued)" : "");

  return `### ${title}`;
}
Then in prepareMessageContent’s pipeline:

ts
export function prepareMessageContent(raw: string): string {
  if (!raw) return "";
  const withHeadings = raw
    .split(/\r?\n/)
    .filter(line => !DATE_LINE_PATTERN.test(line.trim()))
    .map(line => normalizePseudoHeadings(line.trim()))
    .map(line => normalizeHashHeadings(line))      // <— new
    .map(line => normalizeTransitionLines(line))   // your existing --- Now providing ---
    .join("\n");

  return normalizeParagraphs(withHeadings);
}
